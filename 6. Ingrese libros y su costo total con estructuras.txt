//Ingrese libros y su costo total con estructuras
#include <iostream>
#include <string>
using namespace std;
struct texto{
    string nombre;
    float costo;
};

int main (){
    int n,B=0;
    cin>>n;
    texto libro[n];
    for(int i=0;i<n;i++){
        cout<<"Ingrese el nombre del libro: ";
        cin>>libro[i].nombre;
        cout<<"Ingrese el costo del libro: ";
        cin>>libro[i].costo;
        cout<<endl;
    }
    
    for(int i=0;i<n;i++){
        cout<<"\nEl nombre del libro "<<i+1<<" es: "<<libro[i].nombre<<;
        cout<<endl;
        B+=libro[i].costo;
    }
    cout<<"El costo total de todos los libros es: "<<B;
    return 0;
}
