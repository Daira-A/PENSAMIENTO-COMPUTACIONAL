//Media geometrica
#include <iostream>
#include <cmath>
using namespace std;
int main (){
    int n;
    cin>>n;
    double A[n], producto=1,mg;
    for(int i=0;i<n;i++){
        cin>>A[i];
    }
    for(int i=0;i<n;i++){
        producto=producto*A[i];
    }
    mg=pow(producto,1/(double)n);
    cout<<mg;
    return 0;
}
