//Media armonica
#include <iostream>
using namespace std;
int main (){
    float a;
    int n;
    cin>>n;
    int A[n];
    for(int i=0;i<n;i++){
        cin>>A[i];
    }
    a=0;
    for(int i=0;i<n;i++){
        a=a+(1.0/A[i]);
    }
    cout<<"La respuesta es: "<<n/a;
    return 0;
}
